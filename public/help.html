Procedures and functions

A procedure is a group of commands which you can call.
A program must contain at least one procedure which must be called main.

Main procedure

proc main()
endp
Example
/examples/procedures/mainProc.whlp

Procedure pointer

Procedure pointers can be stored in proc typed variables.

Setting the pointer can be done with the set command.
Once a value is assigned the pointer can be called like a normal procedure. 

#project "Array of procedure pointers demo"

#include "/lib/standard.whl"

proc procList[2]

proc demoProc1()
    printS("Procedure 1")
endp

proc demoProc2()
    printS("Procedure 2")
endp

proc main()
    arrayw procList, 0, demoProc1
    arrayw procList, 1, demoProc2
    proc p
    arrayr p, procList, 1
    p()
    arrayr p, procList, 0
    p()
endp

Procedure parameters

Procedures can be called with parameters, all parameters are passed by value.

#project "Number parameter demo"

#include "/lib/standard.whl"

proc printPoint(number x, number y, number z)
    printS("Point:")
    printN(x)
    printN(y)
    printN(z)
endp

proc main()
    printPoint(561, 520, 974)
endp

#project "Struct parameter demo"

#include "/lib/standard.whl"

struct Point
    number x
    number y
    number z
ends

proc printPoint(Point point)
    printS("Point:")
    printN(point.x)
    printN(point.y)
    printN(point.z)
endp

Point point

proc main()
    set point.x, 1656
    set point.y, 98
    set point.z, 75
    printPoint(point)
endp

#project "Array parameter demo"

#include "/lib/standard.whl"

number p[3]

proc printPoint(number point[3])
    printS("Point:")

    number n

    arrayr n, point, 0
    printN(n)
    arrayr n, point, 1
    printN(n)
    arrayr n, point, 2
    printN(n)
endp

proc main()
    arrayw  p, 0, 3975
    arrayw  p, 1, 296
    arrayw  p, 2, 7013
    printPoint(p)
endp
