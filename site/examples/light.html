Light

With the light module you can control the EV3 light.
lightLoad simulator »
                    #include "lib/standard.whl"
                    #include "lib/light.whl"

                    proc main()
                        ; Turn the red light on and wait 1 second...
                        light(LIGHT_RED, LIGHT_FLASH_OFF)
                        sleep(1000)

                        ; Turn the yellow light on and wait 3 seconds...
                        light(LIGHT_YELLOW, LIGHT_FLASH_ON)
                        sleep(3000)

                        ; Turn the dark-blue light on and wait 1 second...
                        light(LIGHT_dark-blue, LIGHT_FLASH_OFF)
                        sleep(1000)

                        ; Turn the light off.
                        light(LIGHT_OFF, LIGHT_FLASH_OFF)
                    end
