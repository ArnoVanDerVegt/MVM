Math

Math operations other than the 5 VM operator commands (add, sub, mul, div, mod).

roundLoad simulator

Round a value to the closest integer value. 
                    #include "lib/math.whl"
                    #include "lib/screen.whl"

                    proc main()
                        number a
                        number b

                        a = 0.6
                        ; b will be a rounded up to 1
                        b = round(a)

                        drawNumber(10, 10, a)
                        drawNumber(10, 30, b)

                        a = 2.4
                        ; b will be a rounded down to 2
                        b = round(a)
                        drawNumber(10, 60, a)
                        drawNumber(10, 80, b)
                    endp

absLoad simulator

Get the absolute value. 
                    #include "lib/math.whl"
                    #include "lib/screen.whl"

                    proc main()
                        number a
                        number b

                        a = -9
                        ; b will be the positive value of a (9)
                        b = abs(a)

                        drawNumber(10, 10, a)
                        drawNumber(10, 30, b)

                        a = 67
                        ; b will be the same value as a (67)
                        b = abs(a)
                        drawNumber(10, 60, a)
                        drawNumber(10, 80, b)
                    endp
