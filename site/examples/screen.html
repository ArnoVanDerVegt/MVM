CircleLoad simulator 

                    #project "Circle drawing demo"

                    #include "lib/standard.whl"
                    #include "lib/screen.whl"

                    proc main()
                        ; Draw a circle at (60, 50) with a radius of 40
                        drawCircle(60, 50, 20)

                        ; Draw black a filled circle
                        setFill(1)
                        drawCircle(70, 60, 20)

                        ; Draw a white filled circle
                        setFillColor(1)
                        drawCircle(80, 60, 10)
                    endp
