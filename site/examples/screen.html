Screen

The screen module provides access to screen drawing functions.
setFill
Set the fill, if 1 then circles and rectangles are filled. If 0 then only lines the outline is drawn.
See: Circle, Buttons, Rect.
setFillColor
Set the fill color. See: Circle, Rect.
setTextSize
Set the text size. See: Circle, Rect.
drawCircleLoad simulator »
                    #include "lib/screen.whl"

                    proc main()
                        ; Draw a circle at (60, 50) with a radius of 40
                        drawCircle(69, 64, 25)

                        ; Draw black a filled circle
                        setFill(1)
                        drawCircle(89, 64, 20)

                        ; Draw a white filled circle
                        setFillColor(1)
                        drawCircle(109, 64, 15)
                    end
drawNumberLoad simulator »
Draw a text with the active fill and color settings. 
                    #include "lib/screen.whl"

                    proc main()
                        number n

                        for n = 0 to 10
                            number x
                            number y

                            x = n
                            x *= 10
                            y = n
                            y *= 4
                            y += 10

                            drawNumber(x, y, n)
                        end
                    end
drawTextLoad simulator »
Draw a text with the active fill and color settings. 
                    #include "lib/screen.whl"

                    proc main()
                        ; Draw a small black text at the left top of the screen...
                        drawText(10, 10, "The quick brown")

                        ; Set the text size to small bold:
                        setTextSize(1)
                        drawText(10, 25, "fox jumped over")

                        ; Set the text size to large:
                        setTextSize(2)
                        drawText(10, 40, "Lazy dogs")

                        ; Set the text size back to small:
                        setTextSize(0)
                        ; Set the fill:
                        setFill(1)
                        ; This draws small text with black bagkround...
                        drawText(10, 60, "The quick brown")
                    end

drawPixelLoad simulator »
The following demo draws a 100 pixels in a grid: 
                    #include "lib/screen.whl"

                    proc main()
                        number i, j
                        number x, y

                        y = 0
                        for i = 0 to 10
                            x = 0
                            for j = 0 to 10
                                drawPixel(x, y)

                                x += 10
                            end

                            y += 10
                        end
                    end
