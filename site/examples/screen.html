Examples

Screen

Circle

#project "Circle drawing demo"

#include "/lib/screen.whl"

proc main()
    ; Draw a circle at (60, 50) with a radius of 40
    drawCircle(60, 50, 40)
endp

Draw text

#project "Draw text demo"

#include "/lib/screen.whl"

proc main()
    drawText(0, 0, "Hello")
endp

Image

#project "Image demo"

#include "/lib/screen.whl"

proc main()
    drawImage(0, 0, "image.rgf")
endp

Line

#project "Line drawing demo"

#include "/lib/screen.whl"

proc main()
    ; Draw a line from (5, 10) to (25, 35)
    drawLine(5, 10, 125, 105)
endp

Pixel

#project "Pixel drawing demo"

#include "/lib/screen.whl"

proc main()

    drawPixel(155, 10)
endp

Rect

#project "Rectangle drawing demo"

#include "/lib/screen.whl"

proc main()
