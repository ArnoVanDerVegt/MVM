drawPixel

An implementation of a call to the pixel drawing api. 
                    #define MODULE_SCREEN     1
                    #define SCREEN_DRAW_PIXEL 4

                    proc drawPixel(number x, number y)
                        struct DrawPixel ; Declare a struct for the parameters
                            number x, y
                        ends
                        DrawPixel drawPixel
                        ; Set the parameter values:
                        set    drawPixel.x, x
                        set    drawPixel.y, y
                        ; Set the source register to the address of the drawPixel record:
                        addr   drawPixel
                        module MODULE_SCREEN, SCREEN_DRAW_PIXEL
                    endp
